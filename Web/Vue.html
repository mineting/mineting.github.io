        <a v-bind:href="Date.now()" :x="info.message">指令语法测试</a>
    </div>
    <script>
        Vue.config.prodcutionTip = false
        new Vue({
            el:'#root',
            data:{
                message: 'World',
                info: {
                    url: 'MineTing',
                    math: 'doublex',
                    message: 'Example'
                }
            }
        })
    </script>
</body>
</html>

                                              # el 与 data 的两种写法

                                              <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>el 与 data 的两种写法</title>
    <script src="./js/vue.js"></script>
</head>
<body>
    <!-- 
        el 与 data 的两种写法：
            el：
                （1）new Vue()时配置 el
                （2）Vue 创建好后挂载 el, vm.$mount('#root');
            data：
                （1）对象式
                （2）函数式（组件使用）
                    注意：不能写成箭头函数，Vue去调用该函数，this 是Vue实例
     -->
    <div id="root">
        <h1>Hello {{message}}!</h1>
    </div>
    <script>
        Vue.config.productionTip = false; 

        const vm = new Vue({
            // el 的第一种写法
            // el: "#root",

            // data的第一种写法：对象式
            // data: {
            //     message:'Vue'
            // }

            // data的第二种写法：函数式
            // data:function(){
            data(){
                console.log(this); //此处的 this 是 Vue 实例对象，不能写成箭头函数，否则 this 是 Window 对象
                return {
                    message: 'Vue'
                }
            }
        });
        //el 第二种写法：更加灵活
        setTimeout(() => {
            vm.$mount('#root');
        }, 1000);
    </script>
</body>
</html>

                                              # Object.defineProperty

                                              <!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Object.defineProperty</title>
    <script src="./js/vue.js"></script>

</head>

<body>
    <!-- 
        由 Object.defineProperty 设置对象的值：
            默认不能被枚举、修改、删除，需要进行配置
            // value: 13,
            // enumerable: true,   //控制属性是否可以被枚举 FALSE
            // writable: true,  //控制属性是否可以被修改，默认为 FALSE
            // configurable: true,  //控制属性是否可以被删除，默认为 FALSE
            
            其中 get 和 set 方法可以实现数据的双向绑定
     -->
    <script>

        let number = 8;
        //
        let person = {
            name: "tom",
            sex: 'Man',
            // age: 12,
            // 此处设置的number只会赋值一次，number改变时，person对象的age属性值不会改变
            age: number
        }

        //注意属性的引号
        Object.defineProperty(person, 'age', {
            // value: 13,
            // enumerable: true,   //控制属性是否可以被枚举 FALSE
            // writable: true,  //控制属性是否可以被修改，默认为 FALSE
            // configurable: true,  //控制属性是否可以被删除，默认为 FALSE

            //读取 person.age 属性值时，会调用该方法，返回值即为age的值
            get: function () {
                console.log("getter....");
                return number;
            },
            //设置 person.age 属性值时，会调用该方法
            set(value) {
                console.log("setter....value: ", value);
                // 绑定number和value
                number = value
            }
        })

        console.log(Object.keys(person));
        for (let key in person) {
            console.log(person[key]);
        }

        console.log(person);
    </script>
</body>

</html>

                                              <!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>数据代理</title>
</head>

<body>
    <!-- 
        数据代理：通过一个对象代理对另一个对象中属性的操作：读写等
     -->
    <script>
        let obj = { x: 10 }
        let obj2 = { y: 20 }
        Object.defineProperty(obj2, 'x', {
            get() {
                return obj.x;
            },
            set(value) {
                obj.x = value;
            }
        })
    </script>
</body>

            vm.data = undefine; Vue 会将 data 中的值存在 _data 中，并额外的在 _data 上通过观察者模式添加与页面的响应

        let data = {
            name: 'Sample',
            address: 'Xian'
        };

            el: "#root",
            data
        });

    </script>
</body>

                                              # 事件的基本使用

                                              <!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>事件的基本使用</title>
    <script src="./js/vue.js"></script>
</head>

<body>
    <!-- 
        事件的基本使用：
        1. 使用 v-on:事件名 或者 @事件名
        2. 时间的回调配置在 methods 上，最终显示在 vm 上
        3. methods 中配置的函数不要使用箭头函数，this 应指向vm
        4. 函数名后可以加或不加括号(默认传递事件对象)，如果传参需要添加括号（不会默认传递事件对象）
            如果在插值语法中使用函数：不加括号会直接返回函数定义信息，添加括号才会得到返回值
     -->
    <div id="root">
        <h1>测试事件的基本使用</h1>
        <!-- 函数名后可以加或不加括号(默认传递事件对象)，如果传参需要添加（不会默认传递事件对象） -->
        <button v-on:click="showInfo1">事件一（不传参）</button>
        <!-- $event：可以看做 Vue 的关键字，表示事件对象，与参数顺序无关 -->
        <button @click="showInfo2(2,$event)">事件二（传参）</button>
    </div>

    <script>
        Vue.config.productionTip = false;

        const vm = new Vue({
            el: "#root",
            //只有data中的属性/函数才会进行数据劫持,生成getter/setter
            data: {
                showInfo3() {

                }
            },
            methods: {
                //注意不要使用箭头函数
                // showInfo1:()=>{}
                showInfo1() {
                    console.log("showInfo1...");
                },
                showInfo2(num, event) {
                    console.log(num, event.target.innerHTML);
                }
            }
        });
        console.log(vm);

    </script>
</body>

</html>

                                              # 事件修饰符

                                              <!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>事件修饰符</title>
    <script src="./js/vue.js"></script>
    <style>
        .demo1 {
            margin: 20px 0 20px;
            height: 50px;
            background-color: aqua;
        }

        .box1 {
            padding: 5px;
            background-color: skyblue;
        }

        .box2 {
            padding: 5px;
            background-color: orange;
        }

        .list {
            width: 200px;
            height: 200px;
            background-color: greenyellow;
            overflow: auto;
        }

        li {
            height: 100px;
        }
    </style>
</head>

<body>
    <!-- 
        Vue 事件修改符：
        1. prevent：阻止默认事件
        2. stop：阻止事件冒泡
        3. once：之间只触发一次
        4. capture：使用事件的捕获模式,在捕获阶段进行事件的执行，而不是在冒泡阶段
        5. self：只有event.target是当前操作元素才会触发事件
        6. passive：事件的默认行为无需等待事件回调执行完毕
     -->
    <div id="root">
        <h1>事件修饰符</h1>
        <!-- 阻止默认行为 -->
        <a href="http://www.baidu.com" @click.prevent="showInfo1">超链接默认修饰符</a>
        <!-- 阻止事件冒泡 -->
        <div class="demo1" @click="showInfo2">
            <button @click.stop="showInfo2">阻止事件冒泡</button>
        </div>
        <!-- 事件只触发一次 -->
        <button @click.once="showInfo1">事件只触发一次</button>
        <!-- 事件执行过程：先捕获再冒泡 -->
        <div class="box1" @click.capture="showMessage(1)">
            box1
            <div class="box2" @click="showMessage(2)">
                box2
            </div>
        </div>

        <!-- self：只有event.target是当前操作元素才会触发事件,类似于阻止事件冒泡 -->
        <div class="demo1" @click.self="showInfo2">
            <button @click.stop="showInfo2">SELF 事件</button>
        </div>

        <!-- passive：事件的默认行为无需等待事件回调执行完毕 -->
        <!-- 
            wheel：滚轮事件
            scroll：滚动条事件
            示例：滚轮事件执行完毕后，滚动条才会滚动，
                使用@wheel.passive会先执行默认行为，再执行滚轮事件
         -->
        <!-- <ul @wheel="demo" class="list"> -->
        <ul @wheel.passive="demo" class="list">
            <li>A</li>
            <li>B</li>
            <li>C</li>
            <li>D</li>
        </ul>
    </div>

    <script>
        Vue.config.productionTip = false;

        const vm = new Vue({
            el: "#root",
            data: {

            },
            methods: {
                showInfo1(e) {
                    // e.preventDefault()
                    console.log("showInfo1...");
                },
                showInfo2(e) {
                    // e.stopPropagation()
                    console.log("showInfo2...");
                },
                showMessage(num) {
                    console.log("message: ", num);
                },
                demo() {
                    for (let index = 0; index < 100000; index++) {
                        console.log("emo");
                    }
                    console.log("End...");
                }

            }
        });
        console.log(vm);

    </script>
</body>

</html>

                                              # 键盘事件

                                              <!DOCTYPE html>
<html>

<head>
    <meta charset="UTF-8" />
    <title>键盘事件</title>
    <script type="text/javascript" src="./js/vue.js"></script>
</head>

<body>
    <!-- 
        1. Vue中常用的按键别名：
                    回车 => enter
                    删除 => delete (捕获“删除”和“退格”键)
                    退出 => esc
                    空格 => space
                    换行 => tab (特殊，必须配合keydown去使用,tab键默认移除焦点)
                    上 => up
                    下 => down
                    左 => left
                    右 => right

        2. Vue未提供别名的按键，可以使用按键原始的key值去绑定，但注意要转为kebab-case（短横线命名）

                    (1).配合 keyup 使用：按下修饰键的同时，再按下其他键，随后释放其他键，事件才被触发。
                    (2).配合 keydown 使用：正常触发事件。

        4. 也可以使用keyCode去指定具体的按键（不推荐）

        5.Vue.config.keyCodes.自定义键名 = 键码，可以去定制按键别名
		-->

    </div>

        },

                                              # 计算属性

                                              <!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>姓名案例</title>
    <script src="./js/vue.js"></script>
</head>

<body>
    <!-- 
        计算属性：
            1. 要用的属性不存在，要通过已有属性计算得来，也是属性
            2. 原理：底层借助了 Objcet.defineproperty 方法提供的 getter 和 setter
            3. get函数执行时机
                        (1).初次读取时会执行一次
                        (2).当依赖的数据发生改变时会被再次调用
            4. 优势：与methods实现相比，内部有缓存机制（复用），效率更高，调试方便
            5. 注意事项：
                    （1）计算属性最终会出现在 vm 上，直接读取使用即可
                    （2）如果计算属性要被修改，那必须写 set 函数去响应修改，且 set 中要引起计算时依赖的数据发生改变
            6. 计算属性的简写形式：
                不考虑修改时，只需要getter
     -->
    <div id="root">
        <h2>计算属性实现</h2>
        姓：<input type="text" v-model:value="firstName" /><br>
        名：<input type="text" v-model="lastName" /><br>
        姓名：<span>{{fullName}}</span>
        <!-- 计算属性会进行缓存 -->
        姓名：<span>{{fullName}}</span>
        姓名：<span>{{fullName}}</span>
    </div>
    <script>
        const vm = new Vue({
            el: "#root",
            data: {
                firstName: "",
                lastName: "",
            },
            computed: {
                // vm._data.fullName = undefined
                fullName: {
                    // 读取 fullName 时，get 的返回值会作为 fullName 的值
                    // get 的调用时机：1. 初次读取；2. 所依赖的属性发生变化
                    // 不要写成箭头函数
                    get() {
                        console.log("fullName get...");
                        // console.log(this);
                        return this.firstName + "-" + this.lastName;
                    },
                    // 当 fullName 被修改时调用,一般不会修改
                    set(value) {
                        console.log("fullName set...");
                        const arr = value.split('-');
                        this.firstName = arr[0]
                        this.lastName = arr[1]
                    }
                },
                
                // 简写形式
                // fullName2:function() {
                fullName2() {
                    return this.firstName + "-" + this.lastName;
                }
            }
        })
    </script>
</body>

</html>
