public class BIOServer {
    public static void main(String[] args) throws Exception {
        ExecutorService threadPool = Executors.newCachedThreadPool();
        ServerSocket serverSocket = new ServerSocket(6000);
        System.out.println("BIOServer start...");
        while (true) {
            System.out.println("Waiting Clients Connect...");
            Socket socket = serverSocket.accept();
            System.out.println("Clients Connect...");
            threadPool.execute(() -> {
                handler(socket);
            });
        }
    }

    public static void handler(Socket socket) {
        System.out.println("Thread Id: " + Thread.currentThread().getId() + ", ThreadName: " + Thread.currentThread().getName());
        byte[] bytes = new byte[1024];
        int read = -1;
        try {
            InputStream in = socket.getInputStream();
            //while ((read = in.read(bytes)) != -1) {}
            while (true) {
                System.out.println("handler will read...");
                read = in.read(bytes);
                if (read != -1) {
                    System.out.println(new String(bytes, 0, read));
                } else {
                    break;
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                socket.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
}
