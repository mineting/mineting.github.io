                                                                                                                                                                                                                                      # 示例一：反转字符串

                                                                                                                                                                                                                                      344. 反转字符串 - 力扣（LeetCode）open in new window

                                                                                                                                                                                                                                      编写一个函数，其作用是将输入的字符串反转过来。输入字符串以字符数组 s 的形式给出。

                                                                                                                                                                                                                                      不要给另外的数组分配额外的空间，你必须原地修改输入数组、使用 O(1) 的额外空间解决这一问题。

                                                                                                                                                                                                                                      示例 1：

                                                                                                                                                                                                                                      输入：s = ["h","e","l","l","o"]
输出：["o","l","l","e","h"]

                                                                                                                                                                                                                                      示例 2：

                                                                                                                                                                                                                                      输入：s = ["H","a","n","n","a","h"]
输出：["h","a","n","n","a","H"]

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 字符串反转
     * @param chars
     * @return
     */
    public static char[] reverseString01(char[] chars) {
        int i = 0, j = chars.length - 1;
        while (i < j) {
            char tmp = chars[i];
            chars[i] = chars[j];
            chars[j] = tmp;
            i++;
            j--;
        }
        return chars;
    }

    public static char[] reverseString02(char[] chars) {
        int i = 0, j = chars.length - 1;
        while (i < j) {
            //寻找公共值
            chars[i] ^= chars[j];
            chars[j] ^= chars[i];
            chars[i] ^= chars[j];
            i++;
            j--;
        }
        return chars;
    }

                                                                                                                                                                                                                                      # 数组

                                                                                                                                                                                                                                      # 基本介绍

                                                                                                                                                                                                                                      数组：数组是存放相同类型数据的集合，一维数组在内存上地址是连续的，下标从 0 开始，二维数组对于 Java 不是连续的。

                                                                                                                                                                                                                                      知道了数组的数据起始地址 Base Address，就可以由公式 BaseAddress + i * size 计算出索引 i 元素的地址

                                                                                                                                                                                                                                      • i 即索引，在 Java，C 等语言都是从 0 开始
                                                                                                                                                                                                                                      • size 是每个元素占用字节，例如 int 占 4，double 占 8

                                                                                                                                                                                                                                      例如，byte[] bytes = {1,2,3,4,5} 的起始地址为 0x27324fc8，则元素 3 的地址为 0x27324fc8 + 2 * 1 = 0x27324fca

                                                                                                                                                                                                                                      Java 中数组的空间占用：

                                                                                                                                                                                                                                      Java 中数组结构为：对象头 + 数据

                                                                                                                                                                                                                                      • 8 字节 markword（哈希码、垃圾回收时的分代年龄、锁的信息等）
                                                                                                                                                                                                                                      • 4 字节 class 指针（类指针，数组元素类型，如 int 数组为 int[].class，压缩 class 指针的情况）
                                                                                                                                                                                                                                      • 4 字节数组大小（决定了数组最大容量是 2^32）
                                                                                                                                                                                                                                      • 数组元素 + 对齐字节（ Java 中所有对象大小都是 8 字节的整数倍 ，不足的要用对齐字节补足）

                                                                                                                                                                                                                                      例如，int[] array = {1, 2，3，4，5}; 的大小为 40 个字节，组成如下：8 + 4 + 4 + 5 * 4 + 4 （alignment）

                                                                                                                                                                                                                                      数组的优点：

                                                                                                                                                                                                                                      随机访问：即根据索引查找元素，时间复杂度为 O(1)

                                                                                                                                                                                                                                      动态数组：支持扩容，内部维护了 size，每次增加或删除元素时，都会移动 size 的位置，当增加的元素大小超过了 size，则会拷贝新的大小为 倍数 * size 的新数组，赋值原数组。

                                                                                                                                                                                                                                      动态数组示例：

                                                                                                                                                                                                                                      package datastructure.array;

import java.util.Arrays;
import java.util.Iterator;
import java.util.Spliterator;
import java.util.function.Consumer;
import java.util.stream.IntStream;

/**
 * 动态数组：
 * 插入或删除性能
 * 头部、中间位置，时间复杂度是O(n)
 * 尾部位置，时间复杂度是0(1)
 */
public class DynamicArray implements Iterable<Integer> {

    // 指针
    private int size = 0;
    // 数组容量
    private int capacity = 8;
    // 数据结构
    // private int[] array = new int[capacity];
    private int[] array = {};

    /**
     * 在数组最后添加元素
     * @param element
     */
    public void addLast(int element) {
        // array[size++] = element;
        add(size, element);
    }

    /**
     * 在数组中指定位置添加元素
     * @param index
     * @param element
     */
    public void add(int index, int element) {
        checkAndGrow();
        /* if (index >= 0 && index < size) {
            System.arraycopy(array, index, array, index + 1, size - index);
            array[index] = element;
            size++;
        } */
        if (index >= 0 && index < size) {
            System.arraycopy(array, index, array, index + 1, size - index);
        }
        // 如果 index 等于 size，则方法和 addLast 相同
        array[index] = element;
        size++;
    }

    /**
     * 数组扩容
     */
    public void checkAndGrow() {
        if (size == 0) {
            array = new int[capacity];
        } else if (size == capacity) {
            capacity += capacity >> 1;
            int[] newArray = new int[capacity];
            System.arraycopy(array, 0, newArray, 0, size);
            array = newArray;
        }
    }

    public int get(int index) {
        return array[index];
    }

    /**
     * 遍历方式一：自定义 forEach 遍历
     * 此处 forEach 与迭代器接口的方法冲突
     * @param consumer 函数式接口
     */
    public void forEach2(Consumer<Integer> consumer) {
        for (int i = 0; i < size; i++) {
            // 此处的操作可以通过参数传递：Consumer 接口，由调用方自定义操作
            // System.out.print(array[i] + " ");
            consumer.accept(array[i]);
        }
    }

    public int size() {
        return size;
    }

    /**
     * 遍历方式一：使用迭代器遍历
     * @return
     */
    @Override
    public Iterator<Integer> iterator() {
        return new Iterator<Integer>() {
            int index = 0;

            // 是否有下一个元素
            @Override
            public boolean hasNext() {
                return index < size;
            }

            // 返回当前元素，并移动到下一个元素
            @Override
            public Integer next() {
                return array[index++];
            }
        };
    }

    @Override
    public void forEach(Consumer<? super Integer> action) {
        Iterable.super.forEach(action);
    }

    @Override
    public Spliterator<Integer> spliterator() {
        return Iterable.super.spliterator();
    }

    /**
     * 遍历方式三：使用 Stream
     */
    public IntStream stream() {
        return IntStream.of(Arrays.copyOfRange(array, 0, size));
    }

    public int remove(int index) {
        int removed = array[index];
        // 如果删除的是最后一个元素，则不用拷贝
        if (index < size - 1) {
            System.arraycopy(array, index + 1, array, index, size - index - 1);
        }
        size--;
        return removed;
    }
}

                                                                                                                                                                                                                                      缓存与局部性原理：

                                                                                                                                                                                                                                      局部性原理（这里只讨论空间局部性）

                                                                                                                                                                                                                                      • CPU 读取内存（速度慢）数据后，会将其放入高速缓存（速度快）当中，如果后来的计算再用到此数据，在缓存中能读到的话，就不需要读内存了

                                                                                                                                                                                                                                      • 缓存的最小存储单位是缓存行（cache line），一般是 64 bytes，因此最少读 64 bytes 填满一个缓存行，因此读入某个数据时也会读取其临近的数据（相邻地址，数组元素的存储也是相邻地址），这就是所谓空间局部性

                                                                                                                                                                                                                                      • 代码示例：

                                                                                                                                                                                                                                        package datastructure.array;

import org.springframework.util.StopWatch;

/**
 * 缓存与局部性原理
 * CPU 计算速度皮秒，内存速度：纳秒，因此需要在两者中间引入高速缓存，充分利用 CPU 资源
 * 内存中的数据每次会读取 64 字节（缓存行 CacheLine）放入缓存， 因此读取数据时，会将当前数据以及邻近的凑够 64 字节数据放入缓存行
 *
 * 因此使用 ij 方法时（先读取每一行时，会把后序地址的数据也读入内存中），读取到第一个数据时，会将后序该行每一列的数据都读入到高速缓存中，不需要从内存中读取数据
 * 使用 ji 方法时（先读取每一列，则每次只会把该数据所在的相邻的地址，读取到高速缓存中，读入到高速缓存中的数据不能得到利用，且缓存中最多 512K[8192Line],
 * 之后再读入的数据会覆盖掉之前的缓存行[CacheLine]，因此每次都需要从内存中重新读取数据）
 *
 */
public class CacheLineTest {
    public static void main(String[] args) {
        int rows = 1_000_000;
        int columns = 14;
        int[][] arr = new int[rows][columns];
        StopWatch stopWatch = new StopWatch();
        stopWatch.start("task ij");
        ij(arr, rows, columns);
        stopWatch.stop();

        stopWatch.start("task ji");
        ji(arr, rows, columns);
        stopWatch.stop();

        System.out.println(stopWatch.prettyPrint());
        // sum: 0
        // sum: 0
        // StopWatch '': running time = 89899700 ns
        // ---------------------------------------------
        // ns         %     Task name
        // ---------------------------------------------
        // 016272800  018%  task ij
        // 073626900  082%  task ji
    }

    public static void ij(int[][] arr, int rows, int columns) {
        long sum = 0L;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < columns; j++) {
                sum += arr[i][j];
            }
        }
        System.out.println("sum: " + sum);
    }

    public static void ji(int[][] arr, int rows, int columns) {
        long sum = 0L;
        for (int j = 0; j < columns; j++) {
            for (int i = 0; i < rows; i++) {
                sum += arr[i][j];
            }
        }
        System.out.println("sum: " + sum);
    }

                                                                                                                                                                                                                                      704. 二分查找 - 力扣（LeetCode）open in new window

                                                                                                                                                                                                                                      给定一个 n 个元素有序的（升序）整型数组 nums 和一个目标值 target ，写一个函数搜索 nums 中的 target，如果目标值存在返回下标，否则返回 -1。

                                                                                                                                                                                                                                      输入: nums = [-1,0,3,5,9,12], target = 9
输出: 4
解释: 9 出现在 nums 中并且下标为 4

                                                                                                                                                                                                                                      示例 2:

                                                                                                                                                                                                                                      输入: nums = [-1,0,3,5,9,12], target = 2
输出: -1
解释: 2 不存在 nums 中因此返回 -1

                                                                                                                                                                                                                                      基本思路：

                                                                                                                                                                                                                                      • 对于有序的且不重复的数组元素查找，可以使用二分查找。
                                                                                                                                                                                                                                      • 定义两个指针 left = 0, right = nums.length，用于缩小查找区间范围
                                                                                                                                                                                                                                      • 定义中间值 mid = (left + right) >> 2或者 mid = left + (right - left) / 2，与目标值进行比较，根据比较结果更新 left 和 right 下标。

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 二分查找
     * @param nums
     * @param target
     * @return
     */
    public static int binarySearch(int[] nums, int target) {
        //边界筛选
        if (target < nums[0] || target > nums[nums.length - 1]) {
            return -1;
        }
        int left = 0, right = nums.length - 1;
        //缩小范围，循环条件
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (target == nums[mid]) {
                return mid;
            }
            if (target > nums[mid]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return -1;
    }

                                                                                                                                                                                                                                      • 考虑正数与负数的情况，因为数组有序，最大值只能在两边产生，定义两个指针，分别指向 i = 0 和 j = nums.length-1

     * @param nums
     * @return

    public static int[] method01(int[] nums) {
        int[] ans = new int[nums.length];
        // p 决定平方最大值的下标
        int p = ans.length - 1;
        //注意：i 和 j 的增加由平方值比较结果决定
        for (int i = 0, j = nums.length - 1; i <= j; ) {
            if (nums[i] * nums[i] < nums[j] * nums[j]) {
                ans[p--] = nums[j] * nums[j];
                j--;
            } else {
                ans[p--] = nums[i] * nums[i];
                i++;
            }
        }

                                                                                                                                                                                                                                      基本思路：

                                                                                                                                                                                                                                      • 滑动窗口属于双指针的一种，定义两个指针，作为窗口的左右边界
                                                                                                                                                                                                                                      • 确定窗口：满足条件（连续数组和大于 target）的范围

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 长度最小的子数组：滑动窗口
     * @param nums
     * @param target
     * @return
     */
    public static int method01(int[] nums, int target) {
        int ans = Integer.MAX_VALUE;
        int start = 0, end = 0;
        int sum = 0;
        while (end < nums.length) {
            sum += nums[end];
            while (sum >= target) {
                ans = Math.min(ans, end - start + 1);
                sum -= nums[start];
                start++;
            }
            end++;
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }

    public static void method01(int n) {

        int[][] ans = new int[n][n];
        int left = 0, right = n - 1, top = 0, bottom = n - 1;
        //终止条件
        while (num <= n * n) {
            for (int i = left; i <= right; i++) {

                ans[bottom][i] = num++;
            }
            bottom--;
            for (int i = bottom; i >= top; i--) {
                ans[i][left] = num++;
            }
            left++;
        }

     * 快乐数
     * @param n
     * @return
     */
    public static boolean isHappyNumber(int n) {
        HashSet<Integer> set = new HashSet<>();
        while (n != 1 && !set.contains(n)) {
            set.add(n);
            n = getNextNumber(n);
        }
        return n == 1;
    }
	
    public static int getNextNumber(int num) {
        int ans = 0;
        while (num > 0) {
            int n = num % 10;
            ans += n * n;
            num /= 10;
        }
        return ans;
    }

        v4.edges = List.of(new Edge(v5, 6));
        v5.edges = List.of();
        v6.edges = List.of(new Edge(v5, 9));

        dfsWithRecursion(v1);   // v1 v3 v4 v5 v6 v2 v1
        dfsWithStack(v1);   // v1 v6 v5 v2 v4 v3
    }


    /**
     * DFS：递归实现
     * @param v
     */
    public static void dfsWithRecursion(Vertex v) {
        v.visited = true;
        System.out.print(v.name + " ");
        for (Edge edge : v.edges) {
            if (!edge.linked.visited) {
                dfsWithRecursion(edge.linked);
            }
        }
    }

    /**
     * DFS：栈实现
     * @param v
     */
    public static void dfsWithStack(Vertex v) {
        LinkedList<Vertex> stack = new LinkedList<>();
        stack.push(v);
        while (!stack.isEmpty()) {
            Vertex pop = stack.pop();
            pop.visited = true;
            System.out.print(pop.name + " ");
            for (Edge edge : pop.edges) {
                if (!edge.linked.visited) {
                    stack.push(edge.linked);
                }
            }
        }
    }
}

            System.out.println(v.name + " : " + v.inDegree);

            if (v.inDegree == 0) {
                queue.offer(v);
            }
        }
        // 3. 队列中不断移除顶点，每移除一个顶点，把它相邻的顶点减一，若减到零则入队
        // 检测是否有环：如果有环，则有环的部分因为入度不为 0，因此不能入队，总数小于 graph 数量

            result.add(poll.name);

                edge.linked.inDegree--;
                if (edge.linked.inDegree == 0) {

                                                                                                                                                                                                                                      # Dijkstra 最短路径

                                                                                                                                                                                                                                      Dijkstra （迪克斯特拉）算法：从一个节点到其他节点的最短路径。

                                                                                                                                                                                                                                      • 将所有项点标记为未访问。创建一个末访问顶点的集合。
                                                                                                                                                                                                                                      • 为每个顶点分配一个临时距离值
                                                                                                                                                                                                                                        • 对于初始项点，将其设置为零
                                                                                                                                                                                                                                        • 对于所有其他顶点，将其设置为无穷大
                                                                                                                                                                                                                                      • 每次选择最小临时距离的未访问顶点，作为新的当前顶点
                                                                                                                                                                                                                                      • 对于当前顶点，遍历其所有未访问的邻居，井更新它们的临时距离为更小
                                                                                                                                                                                                                                        • 例如，1 —> 6 的距离是 14，而 1 —> 3 —> 6 的距离是 11。这时将距离更新为 11
                                                                                                                                                                                                                                        • 否则，将保留上次距离值
                                                                                                                                                                                                                                      • 当前顶点的邻居处理完成后，把它从未访问集合中删除

                                                                                                                                                                                                                                      示例：

                                                                                                                                                                                                                                      初始顶点为 1，自身距离为 0，距离其他各个顶点距离为 Infinity，到顶点 3 的距离为 9，更新顶点 1 到顶点 3 的距离为 9，更新顶点 1 到顶点 2 的距离为 7，更新顶点 1 到顶点 6 的距离为 14，当前顶点 1 的邻居处理完成后，把它从未访问集合中删除；处理顶点 2，顶点 2 到顶点 4 的距离由 Infinity 更新为 7 + 15 = 22，把顶点 2 从未访问的集合移除；对于顶点 3，到顶点 4 的距离为 9 + 11 = 20 < 22，更新顶点 4 的最短距离为 20，同理，顶点 3 到顶点 6 的距离更新为 11，即顶点 6 的最短距离更新为 11，当前顶点 3 的邻居处理完成后，把它从未访问集合中删除；顶点 6 到顶点 5 的距离更新为 11 + 9 = 20；把顶点 6 从未访问集合中删除；顶点 5 的最短距离为 20；顶点 4 到顶点 5 的距离为 20 + 6 = 26 < 20 ，不需要更新顶点 5 的最短距离 20。

 *   - 例如，1 —> 6 的距离是 14，而 1 —> 3 —> 6 的距离是 11。这时将距离更新为 11
 *   - 否则，将保留上次距离值
 * - 当前顶点的邻居处理完成后，把它从未访问集合中删除
 */
public class DijkstraUseList {
    public static void main(String[] args) {
        Vertex v1 = new Vertex("v1");
        Vertex v2 = new Vertex("v2");
        Vertex v3 = new Vertex("v3");
        Vertex v4 = new Vertex("v4");
        Vertex v5 = new Vertex("v5");
        Vertex v6 = new Vertex("v6");
        v1.edges = List.of(new Edge(v3, 9), new Edge(v2, 7), new Edge(v6, 14));
        v2.edges = List.of(new Edge(v4, 15));
        v3.edges = List.of(new Edge(v4, 11), new Edge(v6, 2));
        v4.edges = List.of(new Edge(v5, 6));
        v5.edges = List.of();
        v6.edges = List.of(new Edge(v5, 9));

        List<Vertex> graph = List.of(v1, v2, v3, v4, v5, v6);
        dijkstra(graph, v1);
    }

    /**
     * dijkstra 某一节点到其他节点的最短路径
     * @param graph
     * @param source
     */
    static void dijkstra(List<Vertex> graph, Vertex source) {
        // 将所有项点标记为未访问。创建一个末访问顶点的集合
        ArrayList<Vertex> list = new ArrayList<>(graph);
        // 为每个顶点分配一个临时距离值，对于初始项点，将其设置为零，对于所有其他顶点，将其设置为无穷大
        source.dist = 0;
        while (!list.isEmpty()) {
            // 选取当前节点
            Vertex current = chooseMinDistVertex(list);
            // 更新当前节点邻居的距离
            updateNeighboursDist(current, list);
            // 移除当前节点
            list.remove(current);
            // 标记是否处理过（在集合中）
            current.visited = true;
        }

        for (Vertex vertex : graph) {
            System.out.println(vertex.name + " : " + vertex.dist + " : " + (vertex.prev == null ? "null" : vertex.prev.name));
        }
    }

    private static void updateNeighboursDist(Vertex current, ArrayList<Vertex> list) {
        for (Edge edge : current.edges) {
            Vertex neighbour = edge.linked;
            // if (list.contains(neighbour)) {
            // 使用 visited 替代
            if (!neighbour.visited) {
                int dist = current.dist + edge.weight;
                if (dist < neighbour.dist) {
                    neighbour.dist = dist;
                    neighbour.prev = current;
                }
            }
        }
    }

    private static Vertex chooseMinDistVertex(ArrayList<Vertex> list) {
        Vertex min = list.get(0);
        for (int i = 0; i < list.size(); i++) {
            if (list.get(i).dist < min.dist) {
                min = list.get(i);
            }
        }
        return min;
    }

}

                                                                                                                                                                                                                                      package datastructure.graph;

import java.util.Comparator;
import java.util.List;
import java.util.PriorityQueue;

/**
 * 迪克斯特拉：最短路径算法
 * DijkstraUseList （迪克斯特拉）算法：从一个节点到其他节点的最短路径。
 * 【不适用于负权重的边】
 *
 * - 将所有项点标记为未访问。创建一个【优先级队列】
 * - 为每个顶点分配一个临时距离值
 *   - 对于初始项点，将其设置为零
 *   - 对于所有其他顶点，将其设置为无穷大
 * - 每次选择最小临时距离的未访问顶点，作为新的当前顶点
 * - 对于当前顶点，遍历其所有未访问的邻居，井更新它们的临时距离为更小，【若距离更新则加入队列】
 *   - 例如，1 —> 6 的距离是 14，而 1 —> 3 —> 6 的距离是 11。这时将距离更新为 11
 *   - 否则，将保留上次距离值
 * - 当前顶点的邻居处理完成后，把它从【队列】中删除
 */
public class DijkstraUsePriorityQueue {
    public static void main(String[] args) {
        Vertex v1 = new Vertex("v1");
        Vertex v2 = new Vertex("v2");
        Vertex v3 = new Vertex("v3");
        Vertex v4 = new Vertex("v4");
        Vertex v5 = new Vertex("v5");
        Vertex v6 = new Vertex("v6");
        v1.edges = List.of(new Edge(v3, 9), new Edge(v2, 7), new Edge(v6, 14));
        v2.edges = List.of(new Edge(v4, 15));
        v3.edges = List.of(new Edge(v4, 11), new Edge(v6, 2));
        v4.edges = List.of(new Edge(v5, 6));
        v5.edges = List.of();
        v6.edges = List.of(new Edge(v5, 9));

        List<Vertex> graph = List.of(v1, v2, v3, v4, v5, v6);
        dijkstra(graph, v1);
    }

    /**
     * dijkstra 某一节点到其他节点的最短路径
     * @param graph
     * @param source
     */
    static void dijkstra(List<Vertex> graph, Vertex source) {
        // 优先级队列实现为小顶堆，需要传入比较器
        PriorityQueue<Vertex> queue = new PriorityQueue<>(Comparator.comparingInt(v -> v.dist));
        // 为每个顶点分配一个临时距离值，对于初始项点，将其设置为零，对于所有其他顶点，将其设置为无穷大
        source.dist = 0;
        for (Vertex vertex : graph) {
            queue.offer(vertex);
        }
        while (!queue.isEmpty()) {
            // 选取当前节点
            Vertex current = queue.peek();
            // 更新当前节点邻居的距离
            if (!current.visited) {
                updateNeighboursDist(current, queue);
            }
            // 移除当前节点
            queue.poll();
            // 标记是否处理过（在集合中）
            current.visited = true;
        }

        for (Vertex vertex : graph) {
            System.out.println(vertex.name + " : " + vertex.dist + " : " + (vertex.prev == null ? "null" : vertex.prev.name));
        }
    }

    private static void updateNeighboursDist(Vertex current, PriorityQueue<Vertex> queue) {
        for (Edge edge : current.edges) {
            Vertex neighbour = edge.linked;
            // if (list.contains(neighbour)) {
            // 使用 visited 替代
            if (!neighbour.visited) {
                int dist = current.dist + edge.weight;
                if (dist < neighbour.dist) {
                    neighbour.dist = dist;
                    neighbour.prev = current;
                    queue.offer(neighbour);
                }
            }
        }
    }
}

                                                                                                                                                                                                                                          /**
     * 搜索旋转排序数组：二分法
     * @param nums
     * @param target
     * @return
     */
    public static int search(int[] nums, int target) {
        if (nums.length == 0) {
            return -1;
        }
        if (nums.length == 1) {
            return nums[0] == target ? 0 : -1;
        }
        int left = 0, right = nums.length - 1;
        while (left <= right) {
            int mid = (left + right) >> 1;
            if (nums[mid] == target) {
                return mid;
            }
            if (nums[mid] >= nums[0]) {
                if (nums[0] <= target && target < nums[mid]) {
                    right = mid - 1;
                } else {
                    left = mid + 1;
                }
            } else {
                if (nums[mid] < target && target <= nums[nums.length - 1]) {
                    left = mid + 1;
                } else {
                    right = mid - 1;
                }
            }
        }
        return -1;
    }

                                                                                                                                                                                                                                      利用二分，查找第一个小于 target 的目标位置，第一个大于 target 的目标位置。

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 在排序数组中查找元素的第一个和最后一个位置：二分查找
     * @param nums
     * @param target
     * @return
     */
    public static int[] searchRange(int[] nums, int target) {
        int left = binarySearch(nums, target);
        int right = binarySearch(nums, target+1);
        if (left == nums.length || nums[left] != target) {
            return new int[]{-1, -1};
        }
        return new int[]{left, right - 1};
    }

    /**
     * 二分查找
     * @param nums
     * @param target
     * @return
     */
    public static int binarySearch(int[] nums, int target) {
        int left = 0, right = nums.length;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] >= target) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }

                                                                                                                                                                                                                                      1. 主要思想是从初始访问结点出发，初始访问结点可能有多个邻接结点，深度优先遍历的策略就是首先访问第一个邻接结点，然后再以这个被访问的邻接结点作为初始结点，访问它的第一个邻接结点，也就是说每次都在访问完当前结点后首先访问当前结点的第一个邻接结点。
                                                                                                                                                                                                                                      2. DFS 是一个递归的过程，优先往纵向遍历，而不是对一个结点的所有邻接结点进行横向访问。
                                                                                                                                                                                                                                      3. 树是图的一种特例（树是连通无环的图）前序中序后序都属于深度优先遍历。
                                                                                                                                                                                                                                      4. DFS由两种遍历方式：递归和非递归

                                                                                                                                                                                                                                      104. 二叉树的最大深度 - 力扣（LeetCode）open in new window

                                                                                                                                                                                                                                      给定一个二叉树，找出其最大深度。

                                                                                                                                                                                                                                      二叉树的深度为根节点到最远叶子节点的最长路径上的节点数。

                                                                                                                                                                                                                                      示例：
                                                                                                                                                                                                                                      给定二叉树 [3,9,20,null,null,15,7]，

                                                                                                                                                                                                                                          3
   / \
  9  20
    /  \
   15   7

                                                                                                                                                                                                                                      返回它的最大深度 3 。

                                                                                                                                                                                                                                      方式一：DFS

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 获取二叉树的最大深度：DFS
     * @param root
     * @return
     */
    public static int getMaxDepth(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftDepth = getMaxDepth(root.left) + 1;
        int rightDepth = getMaxDepth(root.right) + 1;
        return Math.max(leftDepth, rightDepth);
    }

    /**
     * 获取二叉树的最小深度：DFS
     * @param root
     * @return
     */
    public static int getMinDepth(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftDepth = getMinDepth(root.left) + 1;
        int rightDepth = getMinDepth(root.right) + 1;
        return Math.max(leftDepth, rightDepth);
    }

                                                                                                                                                                                                                                      102. 二叉树的层序遍历 - 力扣（LeetCode）open in new window

                                                                                                                                                                                                                                      给你二叉树的根节点 root ，返回其节点值的 层序遍历 。 （即逐层地，从左到右访问所有节点）。

                                                                                                                                                                                                                                      示例一：

                                                                                                                                                                                                                                      		3
8				20
			15		  7

                                                                                                                                                                                                                                      输入：root = [3,9,20,null,null,15,7]
输出：[[3],[9,20],[15,7]]

                                                                                                                                                                                                                                      示例 2：

                                                                                                                                                                                                                                      输入：root = [1]
输出：[[1]]

                                                                                                                                                                                                                                      方式一：BFS

                                                                                                                                                                                                                                      注意：每次遍历之前，需要确定层级的节点个数

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 二叉树的层序遍历：BFS
     * @param root
     * @return
     */
    public static List<List<Integer>> levelOrder(TreeNode root) {
        if (root == null) {
            return Arrays.asList();
        }
        List<List<Integer>> bfsList = new ArrayList<>();
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            List<Integer> list = new ArrayList<>();
            for (int i = 0; i < queue.size(); i++) {
                TreeNode poll = queue.poll();
                list.add(poll.value);
                if (poll.left != null) {
                    queue.add(poll.left);
                }
                if (poll.right != null) {
                    queue.add(poll.right);
                }
            }
            bfsList.add(list);
        }
        return bfsList;
    }

    /**
     * 二叉树的层序遍历：DFS
     * @param root
     * @return

    /**
     * DFS
     * @param root
     * @param level 层级
     * @return
     */
    public static void dfs(TreeNode root, int level) {
        if (root == null) {
            return;
        }
        if (dfsLevelOrderList.size() < level + 1) {
            dfsLevelOrderList.add(new ArrayList<>());
        }
        List<Integer> list = dfsLevelOrderList.get(level);
        list.add(root.value);
        dfs(root.left, level + 1);
        dfs(root.right, level + 1);
    }

                                                                                                                                                                                                                                      # 示例一：爬楼梯

                                                                                                                                                                                                                                      70. 爬楼梯 - 力扣（LeetCode）open in new window

                                                                                                                                                                                                                                      假设你正在爬楼梯。需要 n 阶你才能到达楼顶。

                                                                                                                                                                                                                                      每次你可以爬 1 或 2 个台阶。你有多少种不同的方法可以爬到楼顶呢？

                                                                                                                                                                                                                                      示例 1：

                                                                                                                                                                                                                                      输入：n = 2
输出：2
解释：有两种方法可以爬到楼顶。

1. 1 阶 + 1 阶
2. 2 阶

                                                                                                                                                                                                                                      示例 2：

                                                                                                                                                                                                                                      输入：n = 3
输出：3
解释：有三种方法可以爬到楼顶。

1. 1 阶 + 1 阶 + 1 阶
2. 1 阶 + 2 阶
3. 2 阶 + 1 阶

                                                                                                                                                                                                                                      提示：

                                                                                                                                                                                                                                      1 <= n <= 45

                                                                                                                                                                                                                                      方式一：递归

                                                                                                                                                                                                                                      解题步骤：

                                                                                                                                                                                                                                      1. 抽象问题的过程，确定方法含义；F(n) 表示爬到当前楼梯 n 所需要走的方法总数。

                                                                                                                                                                                                                                      2. 寻找递推公式；F(n) = F(n-1) + F(n-2)，爬上 n 层楼的方法总数是由爬上 n - 1 和 n -2 层的方法总数决定的。

                                                                                                                                                                                                                                      3. 确定终止条件；由于函数不断的拆分为子问题，终止条件为 F(0) = 1,F(1) = 1，由于递推公式出现了 n - 2，因此需要确

                                                                                                                                                                                                                                        定 F(0) 和 F(1)。

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 递归思想
     * @param n
     * @return
     */
    public static int climbStairs01(int n) {
        //终止条件
        if (n == 0 || n == 1) {
            return 1;
        }
        //递归公式
        return climbStairs01(n - 1) + climbStairs01(n - 2);
    }

                                                                                                                                                                                                                                      方式二：动态规划

                                                                                                                                                                                                                                      由于使用递归计算时，存在两个问题：（1）栈溢出问题；（2）中间数据重复计算的问题

                                                                                                                                                                                                                                      对于 F(n) = F(n-1) + F(n-2) 会有重复计算的问题，因此可以使用 一个数组保存中间结果，使用循环避免递归调用。

                                                                                                                                                                                                                                      解题步骤：

                                                                                                                                                                                                                                      理解题意：每一个状态都是由上一个状态推导得出，考虑动态规划。

                                                                                                                                                                                                                                      • 第一步：确定数组含义，dp[i] 表示爬到当前楼梯所需要走的方法总数

                                                                                                                                                                                                                                      • 第二步：确定状态方程，由于每次可以爬 1 或 2 个台阶，因此爬到当前的方法数等于从下一个楼梯爬上来的方法数加

                                                                                                                                                                                                                                        ​ 上从下两个楼梯爬上来的方法数；便有状态方程：dp[i] = dp[i-1] + dp[i-2]，公式等同于斐波那契数列，但思想不同。

                                                                                                                                                                                                                                      • 第三步：确定初始值，首先初始值是确定的，不能从第二阶开始算，因为爬到第二阶有两种方式，因此初始值需要确

                                                                                                                                                                                                                                        ​ 定第 0 阶和第 1 阶；dp[0] = 1, dp[1] = 1。

                                                                                                                                                                                                                                      代码实现：

                                                                                                                                                                                                                                          /**
     * 动态规划
     * @param n
     * @return
     */
    public static int climbStairs02(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 1;
        dp[1] = 1;
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }

                                                                                                                                                                                                                                      代码实现：此处使用滚动数组的思想

                                                                                                                                                                                                                                          /**
     * 动态规划：滑动数组思想
     * @param n
     * @return
     */
    public static int climbStairs03(int n) {
        int p = 0, q = 0, r = 1;
        for (int i = 1; i <= n; ++i) {
            p = q;
            q = r;
            r = p + q;
        }
        return r;
    }

                                                                                                                                                                                                                                      # 示例二：路径总和

                                                                                                                                                                                                                                      112. 路径总和 - 力扣（LeetCode）open in new window

                                                                                                                                                                                                                                      给你二叉树的根节点 root 和一个表示目标和的整数 targetSum 。判断该树中是否存在 根节点到叶子节点 的路径，这条路径上所有节点值相加等于目标和 targetSum 。如果存在，返回 true ；否则，返回 false 。

                                                                                                                                                                                                                                      叶子节点 是指没有子节点的节点。

                                                                                                                                                                                                                                      示例一：

                                                                                                                                                                                                                                      输入：root = [5,4,8,11,null,13,4,7,2,null,null,null,1], targetSum = 22
输出：true
说明：存在路径 5 --> 4 --> 11 --> 2，使得路径和为 22

                                                                                                                                                                                                                                      示例二：

                                                                                                                                                                                                                                      输入：root = [1,2,3], targetSum = 5
输出：false
解释：树中存在两条根节点到叶子节点的路径：
(1 --> 2): 和为 3
(1 --> 3): 和为 4
不存在 sum = 5 的根节点到叶子节点的路径

                                                                                                                                                                                                                                      解题思路：

                                                                                                                                                                                                                                      询问是否存在从当前节点 root 到叶子节点的路径，满足其路径和为 sum。

                                                                                                                                                                                                                                      假定从根节点到当前节点的值之和为 val，我们可以将这个大问题转化为一个小问题：是否存在从当前节点的子节点到叶子的路径，满足其路径和为 sum - val。

                                                                                                                                                                                                                                      不难发现这满足递归的性质，若当前节点就是叶子节点，那么我们直接判断 sum 是否等于 val 即可（因为路径和已经确定，就是当前节点的值，我们只需要判断该路径和是否满足条件）。若当前节点不是叶子节点，我们只需要递归地询问它的子节点是否能满足条件即可。

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 路径总和：递归实现
     * @param node
     * @param target
     * @return
     */
    public static boolean hasPathSum(TreeNode node, int target) {
        if (node == null) {
            return false;
        }
        if (node.left == null && node.right == null) {
            return node.value == target;
        }
        return hasPathSum(node.left, target - node.value)
                || hasPathSum(node.right, target - node.value);
    }

                                                                                                                                                                                                                                      出现次数最多的子树元素和open in new window

                                                                                                                                                                                                                                      给你一个二叉树的根结点 root ，请返回出现次数最多的子树元素和。如果有多个元素出现的次数相同，返回所有出现次数最多的子树元素和（不限顺序）。

                                                                                                                                                                                                                                      一个结点的 「子树元素和」 定义为以该结点为根的二叉树上所有结点的元素之和（包括结点本身）。

                                                                                                                                                                                                                                      示例一：

                                                                                                                                                                                                                                      输入: root = [5,2,-3]
输出: [2,-3,4]

                                                                                                                                                                                                                                      示例二：

                                                                                                                                                                                                                                      输入: root = [5,2,-5]
输出: [2]

                                                                                                                                                                                                                                      方法一：深度优先搜索
                                                                                                                                                                                                                                      可以从根结点出发，深度优先搜索这棵二叉树。对于每棵子树，其子树元素和等于子树根结点的元素值，加上左子树的元素和，以及右子树的元素和。

                                                                                                                                                                                                                                      用哈希表统计每棵子树的元素和的出现次数，计算出现次数的最大值 maxCnt，最后将出现次数等于 maxCnt 的所有元素和返回。

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          static Map<Integer, Integer> map = new HashMap<>();
    static int maxCount = 0;

    /**
     * 返回出现频次最高的子树和
     * @param root
     * @return
     */
    public static int[] findFrequentTreeSum(TreeNode root) {
        treeSumByDfs(root);
        List<Integer> list = new ArrayList<>();
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            if (entry.getValue() == maxCount) {
                list.add(entry.getKey());
            }
        }
        //返回结果
        int[] ans = new int[list.size()];
        for (int i = 0; i < ans.length; i++) {
            ans[i] = list.get(i);
        }
        return ans;
    }

    /**
     * DFS求子树和
     * @param root
     * @return
     */
    public static int treeSumByDfs(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int sum = root.value + treeSumByDfs(root.left) + treeSumByDfs(root.right);
        map.put(sum, map.getOrDefault(sum, 0) + 1);
        //顺便求出最大次数
        maxCount = Math.max(maxCount, map.get(sum));
        return sum;
    }

                                                                                                                                                                                                                                          /**
     * 合并两个有序列表
     * @param a
     * @param b
     * @return
     */
    public static ListNode mergeTwoList(ListNode a, ListNode b) {
        if (a == null || b == null) {
            return a != null ? a : b;
        }
        ListNode head = new ListNode();
        ListNode tail = head, aPtr = a, bPtr = b;
        while (aPtr != null && bPtr != null) {
            if (aPtr.value < bPtr.value) {
                tail.next = aPtr;
                aPtr = aPtr.next;
            } else {
                tail.next = bPtr;
                bPtr = bPtr.next;
            }
            tail = tail.next;
        }
        tail.next = aPtr != null ? aPtr : bPtr;
        return head.next;
    }

        ListNode tail = head, aPtr = a, bPtr = b;
        while (aPtr != null && bPtr != null) {
            if (aPtr.value < bPtr.value) {
                tail.next = aPtr;
                aPtr = aPtr.next;
            } else {
                tail.next = bPtr;
                bPtr = bPtr.next;
            }
            tail = tail.next;
        }
        tail.next = aPtr != null ? aPtr : bPtr;
        return head.next;
    }

    public static ListNode merge(ListNode[] lists, int left, int right) {
        if (left == right) {
            return lists[left];
        }
        if (left > right) {
            return null;
        }
        int mid = (left + right) >> 1;
        return mergeTwoList(merge(lists, left, mid), merge(lists, mid + 1, right));
    }
    /**
     * 合并两个有序列表
     * @param a
     * @param b
     * @return
     */
    public static ListNode mergeTwoList(ListNode a, ListNode b) {
        if (a == null || b == null) {
            return a != null ? a : b;
        }
        ListNode head = new ListNode();
        ListNode tail = head, aPtr = a, bPtr = b;
        while (aPtr != null && bPtr != null) {
            if (aPtr.value < bPtr.value) {
                tail.next = aPtr;
                aPtr = aPtr.next;
            } else {
                tail.next = bPtr;
                bPtr = bPtr.next;
            }
            tail = tail.next;
        }
        tail.next = aPtr != null ? aPtr : bPtr;
        return head.next;
    }

                                                                                                                                                                                                                                          /**
     * 方式二：二分查找
     * @param matrix
     * @param target
     * @return
     */
    public static boolean searchMatrix02(int[][] matrix, int target) {
        for (int[] row : matrix) {
            int index = search(row, target);
            if (index > 0) {
                return true;
            }
        }
        return false;
    }

    public static int search(int[] row, int target) {
        int low = 0, high = row.length - 1;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            int num = row[mid];
            if (num == target) {
                return mid;
            } else if (target > num) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return -1;
    }

                                                                                                                                                                                                                                      力扣：https://leetcode.cn/problems/fibonacci-number/open in new window

                                                                                                                                                                                                                                      斐波那契数 （通常用 F(n) 表示）形成的序列称为 斐波那契数列 。该数列由 0 和 1 开始，后面的每一项数字都是前面两项数字的和。也就是：

                                                                                                                                                                                                                                      F(0) = 0，F(1) = 1
                                                                                                                                                                                                                                      F(n) = F(n - 1) + F(n - 2)，其中 n > 1
                                                                                                                                                                                                                                      给定 n ，请计算 F(n) 。

                                                                                                                                                                                                                                      示例 1：

                                                                                                                                                                                                                                      输入：n = 2
                                                                                                                                                                                                                                      输出：1
                                                                                                                                                                                                                                      解释：F(2) = F(1) + F(0) = 1 + 0 = 1
                                                                                                                                                                                                                                      示例 2：

                                                                                                                                                                                                                                      输入：n = 3
                                                                                                                                                                                                                                      输出：2
                                                                                                                                                                                                                                      解释：F(3) = F(2) + F(1) = 1 + 1 = 2

                                                                                                                                                                                                                                      【方案一：递归】

                                                                                                                                                                                                                                      递归的思路是最清晰的。

                                                                                                                                                                                                                                      代码实现：

                                                                                                                                                                                                                                          /**
     * 递归实现 F(n) = F(n-1) + F(n-2)
     * 缺点：
     *  1. 导致栈溢出
     *  2. 重复计算
     * @param n
     * @return
     */
    public static int fibonacci01(int n) {
        if (n < 2) {
            return n;
        }
        return fibonacci01(n - 1) + fibonacci01(n - 2);
    }

                                                                                                                                                                                                                                      递归的缺点：（1）栈溢出 （2）重复计算

                                                                                                                                                                                                                                      子问题出现了重复计算，采用动态规划改进。

                                                                                                                                                                                                                                      【方案二：动态规划】

                                                                                                                                                                                                                                      步骤：

                                                                                                                                                                                                                                      • 第一步：确定数组含义，dp[i] 表示当前的斐波那契数值
                                                                                                                                                                                                                                      • 第二步：确定状态方程，F(n) = F(n-1) + F(n-2)
                                                                                                                                                                                                                                      • 第三步：确定初始值，F(0) = 0，F(1) = 1

                                                                                                                                                                                                                                      代码实现：

                                                                                                                                                                                                                                          /**
     * 动态规划
     * @param n
     * @return
     */
    public static int fibonacci02(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }

                                                                                                                                                                                                                                      【方案三：滚动数组思想】

                                                                                                                                                                                                                                      由于F(n)只与F(n-1) 和 F(n-2)有关，且是一维数组，采用滚动数组思想

                                                                                                                                                                                                                                      代码实现：

                                                                                                                                                                                                                                          /**
     * 进一步简化：
     * 由于F(n)只与F(n-1) 和 F(n-2)有关，且是一维数组，采用滚动数组思想
     * @param n
     * @return
     */
    public static int fibonacci03(int n) {
        int p = 0, q = 0, r = 1;
        for (int i = 2; i <= n; i++) {
            p = q;
            q = r;
            r = p + q;
        }
        return r;
    }

                                                                                                                                                                                                                                      把m个同样的苹果放在n个同样的盘子里，允许有的盘子空着不放，问共有多少种不同的分法？

                                                                                                                                                                                                                                      注意：如果有7个苹果和3个盘子，（5，1，1）和（1，5，1）被视为是同一种分法。

                                                                                                                                                                                                                                      数据范围：0≤m≤10 ，1≤n≤10 。

                                                                                                                                                                                                                                      # 输入描述：

                                                                                                                                                                                                                                      输入两个int整数

                                                                                                                                                                                                                                      # 输出描述：

                                                                                                                                                                                                                                      输出结果，int型

                                                                                                                                                                                                                                      # 示例1

                                                                                                                                                                                                                                      输入：

                                                                                                                                                                                                                                      7 3

                                                                                                                                                                                                                                      复制

                                                                                                                                                                                                                                      输出：

                                                                                                                                                                                                                                      8

                                                                                                                                                                                                                                      问题分析：

                                                                                                                                                                                                                                      # 动态规划

                                                                                                                                                                                                                                      使用动态规划思想求解，记 dp[i][j] 表示使用 i 个苹果放入到 j 个盘子的方案数。

                                                                                                                                                                                                                                      1. 若i<j，即苹果比盘子数小
                                                                                                                                                                                                                                        • 比如将 2 个苹果放入 3 个盘子，必然会有一个盘子为空，即{0,x,y}，其中一位固定为 0，不会影响最终的方案数，这和将 2 个苹果放入 2 个盘子的方案数目是一样的，即 {x,y}
                                                                                                                                                                                                                                        • 因此有当 i<j 时，dp[i][j] = dp[i][i]
                                                                                                                                                                                                                                      2. 若i>=j，即苹果数目不小于盘子数时，考虑是否允许有空盘子（比如将 4 个苹果放入 3 个盘子）
                                                                                                                                                                                                                                        • 若允许有空盘，相当于盘子数量少一个（因为固定认为是空盘，不会影响最终方案数目），即 {x,y,0}，这对应着 dp[i][j-1]
                                                                                                                                                                                                                                        • 若不允许有空盘，对于 j 个盘子，相当于拿出来 j 个苹果，分配到每个盘子中，保证每个盘子分配到一个苹果，这对应着 dp[i−j][j]
                                                                                                                                                                                                                                        • 因此有当 i>=j 时，dp[i][j] = dp[i][j-1] + dp[i−j][j]
                                                                                                                                                                                                                                      3. 边界条件
                                                                                                                                                                                                                                        • 对于只有 1 个盘子的情况，分配方案只有 1 种，dp[i][1] = 1;
                                                                                                                                                                                                                                        • 对于只有 0 个苹果的情况，分配方案只有 1 种，dp[0][j] = 1;
                                                                                                                                                                                                                                        • 对于只有 1 个苹果的情况，分配方案只有 1 种，dp[1][j] = 1;

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 方式一：动态规划
     * @param apple
     * @param dish
     * @return
     */
    public static int putApple(int apple, int dish) {
        if (apple == 0 || dish == 0 || apple == 1) {
            return 1;
        }
        //dp[i][j] 表示使用 i 个苹果放入到 j 个盘子的方案数
        int[][] dp = new int[apple + 1][dish + 1];
        //初始化
        for (int i = 0; i <= apple; i++) {
            //盘子数量为1时
            dp[i][1] = 1;
        }
        for (int i = 0; i <= dish; i++) {
            //苹果数为0和1
            dp[0][i] = 1;
            dp[1][i] = 1;
        }
        for (int i = 2; i <= apple; i++) {
            for (int j = 2; j <= dish; j++) {
                if (i < j) {
                    //苹果数量小于盘子数量
                    dp[i][j] = dp[i][i];
                } else {
                    //如果允许有空盘
                    dp[i][j] = dp[i][j - 1];
                    //注意 >=
                    if (i - j >= 0) {
                        //每个盘子都放
                        dp[i][j] += dp[i - j][j];
                    }
                }
            }
        }
        return dp[apple][dish];
    }

                                                                                                                                                                                                                                      # 0-1背包

                                                                                                                                                                                                                                      ​ 有 N 件物品和一个容量为 V 的背包。第 i 件物品的重量是 w[i]，价值是 v[i]。求解将哪些物品装入背包可使这些物品的重量总和不超过背包容量，且价值总和最大。每种物品仅有一件，可以选择放或不放。

                                                                                                                                                                                                                                      # 解题步骤

                                                                                                                                                                                                                                      （1）放入第 i 件物品时的价值依赖于放入第 i-1 件物品时的价值，属于子问题，因此考虑动态规划；

                                                                                                                                                                                                                                      （2）由于每次放入物品时，都要考虑重量，因此使用二维数组 dp[i] [j] 表示第 i 个物品，在容量为 j 时的价值；

                                                                                                                                                                                                                                      （3）每次放入物品时需要考虑放得下还是放不下的问题，针对两种情况判断最优解。

                                                                                                                                                                                                                                      使用表格演示过程：

                                                                                                                                                                                                                                      • 第一步：确定数组含义；使用 dp[i] [j] 表示在容量为 j 时，放入第 i 个物品的总价值。

                                                                                                                                                                                                                                      • 第二步：确定状态方程；此处需要考虑放得下还是放不下的问题。

                                                                                                                                                                                                                                        ​ 放不下：背包的最大价值便是已放入物品的最大价值，dp[i] [j] = dp[i-1] [j]

                                                                                                                                                                                                                                        ​ 放得下：最大价值为当前的物品价值，加上总容量减去当前物品的重量后的最大价值，也就是

                                                                                                                                                                                                                                        ​ dp[i] [j] = v[i] + dp[i - 1] [j - w[i]]

                                                                                                                                                                                                                                        ​ 最大价值取两者的最大值，因此状态方程为：dp[i] [j] = Max{dp[i-1] [j], v[i] + dp[i - 1] [j - w[i]]}

                                                                                                                                                                                                                                      • 第三步：确定初始值；涉及到 i -1 的问题，因此 i = 0时需要初始化

                                                                                                                                                                                                                                      # 示例一：最长上升子序列

                                                                                                                                                                                                                                      子序列是由数组派生而来的序列，删除（或不删除）数组中的元素而不改变其余元素的顺序。

                                                                                                                                                                                                                                      解题思路：需要双重 for 循环，从前往后依次判断第 i 个位置之前的最长上升子序列时（可以二分法优化），判断 j 位置的元素大小，如果数组 nums[i] > nums[j]，则上升子序列的长度 +1，否则保持不变，因此使用一维数组 dp[i] 记录 i 位置之前的上升子序列长度，最终 dp[i] 的最大值即为最长上升子序列的长度。

                                                                                                                                                                                                                                      • 第一步：确定数组含义；dp[i] 记录 i 位置之前的上升子序列长度。
                                                                                                                                                                                                                                      • 第二步：确定状态方程；dp[i] = Math.max(dp[j], dp[j]+1)。
                                                                                                                                                                                                                                      • 第三步：确定初始值；初始每个位置的上升长度都为 1。

                                                                                                                                                                                                                                      方式一：动态规划

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 最长上升子序列
     * @param nums
     * @return
     */
    public static int method01(int[] nums) {
        //数组含义：i 位置之前的最大上升子序列长度
        int[] dp = new int[nums.length];

        //初始值默认为 1
        Arrays.fill(dp, 1);

        //最终dp[i]的最大值即为最长上升子序列的长度
        int ans = 0;
        //i 从 1 开始
        for (int i = 1; i < nums.length; i++) {
            //不断更新 dp[i] 的值，注意 j < i
            for (int j = 0; j < i; j++) {
                //递推公式：nums[i] > nums[j]，则上升子序列的长度 +1，否则保持不变
                if (nums[i] > nums[j]) {
                    //当前位置的最大子序列长度取决于前一个位置的最大上升子序列长度
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                }
            }
            //每次循环确定 dp[i] 的最大值
            ans = Math.max(ans, dp[i]);
        }
        return ans;
    }

                                                                                                                                                                                                                                      # 示例二：合唱队

                                                                                                                                                                                                                                      方式一：动态规划

                                                                                                                                                                                                                                      在基本最长递增子序列问题上加了约束条件，需要左边递增，右边递减。

                                                                                                                                                                                                                                      基本思路：先找到位置 i 之前的最长递增序列 dpLeft[i]，再找到位置 i 之后的最长递减序列 dpRight[i]，求出所有位置的最长递增和最长递减的长度（dpLeft[i] + dpRight[i] -1），最后减去最长的长度即为出队人数。

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 合唱队：动态规划
     * @param nums
     * @return
     */
    public static int method01(int[] nums) {
        //最长左递增高序列 dpLeft[i]
        int[] dpLeft = new int[nums.length];
        //初始化默认为 1
        Arrays.fill(dpLeft, 1);
        for (int i = 1; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[i] > nums[j]) {
                    dpLeft[i] = Math.max(dpLeft[i], dpLeft[j] + 1);
                }
            }
        }
        //最长右递减序列 dpRight[i]，逆序遍历
        int[] dpRight = new int[nums.length];
        //初始化默认为 1
        Arrays.fill(dpRight, 1);

        for (int i = nums.length - 2; i >= 0; i--) {
            for (int j = nums.length - 1; j > i; j--) {
                if (nums[i] > nums[j]) {
                    dpRight[i] = Math.max(dpRight[i], dpRight[j] + 1);
                }
            }
        }
        int[] dpSum = new int[nums.length];
        for (int i = 0; i < dpSum.length; i++) {
            //注意减 1
            dpSum[i] = dpLeft[i] + dpRight[i] - 1;
        }
        //默认最小值为 dp 默认值
        int ans = 1;
        for (int i = 0; i < dpSum.length; i++) {
            ans = Math.max(ans, dpSum[i]);
        }
        return nums.length - ans;
    }

     * @param nums
     * @return
     */
    public static int method01(int[] nums) {
        //确定数组含义
        int[] dp = new int[nums.length];
        Arrays.fill(dp, 1);
        int ans = 0;
        for (int i = 1; i < nums.length; i++) {
            //如果当前值大于前一个值，则 dp[i] + 1，否则重新开始
            if (nums[i] > nums[i - 1]) {
                dp[i] = dp[i - 1] + 1;
            }
            ans = Math.max(ans, dp[i]);
        }
        return ans;

                                                                                                                                                                                                                                      方式一：动态规划

                                                                                                                                                                                                                                      基本思路：

                                                                                                                                                                                                                                      • 确定 dp 数组含义；dp[i] [j]表示字符串 A 的前 i - 1 个字符与字符串 B 的前 j - 1 个字符相同所需要的编辑距离

                                                                                                                                                                                                                                      • 确定递推公式；分为四种情况：

                                                                                                                                                                                                                                        • A[i] == B[j] 时，不用编辑任何距离，dp[i] [j] = dp[i-1] [j-1]，当 A[i] != B[j] 时，分为三种：
                                                                                                                                                                                                                                        • 字符串 A 删除一个元素，dp[i] [j] = dp[i-1] [j] + 1
                                                                                                                                                                                                                                        • 字符串 A 增加一个元素，想当于字符串 B 删除一个元素，dp[i] [j] = dp[i] [j-1] + 1
                                                                                                                                                                                                                                        • 替换元素，dp[i] [j] = dp[i-1] [j-1]

                                                                                                                                                                                                                                      • 因此递推公式为：

                                                                                                                                                                                                                                            if (A[i - 1] == B[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
    }else {
        dp[i][j] = Math.min(dp[i - 1][j - 1], dp[i - 1][j], dp[i][j - 1]) + 1;
    }

                                                                                                                                                                                                                                      • 确定初始值；dp[i] [0] 表示一个字符串为空，则需要的步骤为 i ，同理 dp[0] [j] = j

                                                                                                                                                                                                                                      代码实现：

                                                                                                                                                                                                                                          /**
     * 编辑距离：动态规划
     * @param word1
     * @param word2
     * @return
     */
    public static int method01(String word1, String word2) {
        int m = word1.length();
        int n = word2.length();
        int[][] dp = new int[m + 1][n + 1];
        //初始化
        for (int i = 0; i < m; i++) {
            dp[i][0] = i;
        }
        for (int j = 0; j < n; j++) {
            dp[0][j] = j;
        }
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]) + 1;
                }
            }
        }
        return dp[m][n];
    }

                                                                                                                                                                                                                                      方式二：单调栈

                                                                                                                                                                                                                                      可以维护一个存储下标的单调栈，从栈底到栈顶的下标对应的温度列表中的温度依次递减。如果一个下标在单调栈里，则表示尚未找到下一次温度更高的下标。

                                                                                                                                                                                                                                      正向遍历温度列表。对于温度列表中的每个元素 temperatures[i]，如果栈为空，则直接将 i 进栈，如果栈不为空，则比较栈顶元素 prevIndex 对应的温度 temperatures[prevIndex] 和当前温度 temperatures[i]，如果 temperatures[i] > temperatures[prevIndex]，则将 prevIndex 移除，并将 prevIndex 对应的等待天数赋为 i - prevIndex，重复上述操作直到栈为空或者栈顶元素对应的温度小于等于当前温度，然后将 i 进栈。

                                                                                                                                                                                                                                      为什么可以在弹栈的时候更新 ans[prevIndex] 呢？因为在这种情况下，即将进栈的 i 对应的 temperatures[i] 一定是 temperatures[prevIndex] 右边第一个比它大的元素，试想如果 prevIndex 和 i 有比它大的元素，假设下标为 j，那么 prevIndex 一定会在下标 j 的那一轮被弹掉。

                                                                                                                                                                                                                                      由于单调栈满足从栈底到栈顶元素对应的温度递减，因此每次有元素进栈时，会将温度更低的元素全部移除，并更新出栈元素对应的等待天数，这样可以确保等待天数一定是最小的。

                                                                                                                                                                                                                                      以下用一个具体的例子帮助读者理解单调栈。对于温度列表

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 每日温度：单调栈示例
     * @param temperatures
     * @return
     */
    public static int[] dailyTemperatures(int[] temperatures) {
        int[] ans = new int[temperatures.length];
        Deque<Integer> stack = new LinkedList<>();
        for (int i = 0; i < ans.length; i++) {
            int temperature = temperatures[i];
            //弹栈
            while (!stack.isEmpty() && temperature > temperatures[stack.peek()]) {
                int ele = stack.pop();
                ans[ele] = i - ele;
            }
            //入栈
            stack.push(i);
        }
        return ans;
    }

     * 下一个更大的元素：单调栈
     * @param nums1
     * @param nums2
     * @return
     */
    public static int[] nextGreaterElement(int[] nums1, int[] nums2) {
        Map<Integer, Integer> map = new HashMap<>();
        Deque<Integer> stack = new ArrayDeque<>();
        int[] ans = new int[nums1.length];
        for (int i = nums2.length - 1; i >= 0; --i) {
            int num = nums2[i];
            //弹栈
            while (!stack.isEmpty() && num >= stack.peek()) {
                stack.pop();
            }
            //赋值，注意stack.isEmpty()
            map.put(num, stack.isEmpty() ? -1 : stack.peek());
            //入栈
            stack.push(num);
        }
        //转换
        for (int i = 0; i < nums1.length; ++i) {
            ans[i] = map.get(nums1[i]);
        }
        return ans;
    }

    public static int[] nextGreaterElement(int[] nums) {

        int[] ans = new int[len];
        //初始化为-1
        Arrays.fill(ans, -1);
        Deque<Integer> stack = new LinkedList<>();

            }
            //入栈
            stack.push(i % len);
        }
        return ans;
    }

     * 接雨水：单调栈
     * @param height
     * @return
     */
    public static int trap(int[] height) {
        int ans = 0;
        Deque<Integer> stack = new LinkedList<>();
        for (int i = 0; i < height.length; i++) {
            //弹栈
            while (!stack.isEmpty() && height[i] > height[stack.peek()]) {
                int top = stack.pop();
                //边界处理
                if (stack.isEmpty()) {
                    break;
                }
                int left = stack.peek();
                int currentWith = i - left - 1;
                int currentHeight = Math.min(height[left], height[i]) - top;
                ans += currentHeight * currentWith;
            }
            //入栈
            stack.push(i);
        }
        return ans;
    }

                                                                                                                                                                                                                                      示例二：

                                                                                                                                                                                                                                      输入：isConnected = [[1,0,0],[0,1,0],[0,0,1]]
输出：3

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 省份数量：并查集
     * @param isConnected
     * @return
     */
    public static int findCircleNum(int[][] isConnected) {
        if (isConnected == null || isConnected.length == 0) {
            return 0;
        }
        //初始化
        Union union = new Union(isConnected.length);
        for (int i = 0; i < isConnected.length; i++) {
            for (int j = 0; j < isConnected.length; j++) {
                if (isConnected[i][j] == 1) {
                    union.merge(i, j);
                }
            }
        }
        return union.getCount();
    }

    static class Union {
        int[] root;
        int[] rank;
        int count;

        public Union(int size) {
            root = new int[size];
            rank = new int[size];
            count = size;
            for (int i = 0; i < size; i++) {
                //默认每个元素的父节点就是它本身
                root[i] = i;
                //默认每个元素的深度（秩）为1
                rank[i] = 1;
            }
        }

        /**
         * 查找父节点
         * @param x
         * @return
         */
        int find(int x) {
            if (x == root[x]) {
                return x;
            } else {
                return find(root[x]);
            }
        }

        void merge(int x, int y) {
            int rootX = find(x);
            int rootY = find(y);
            if (rootX != rootY) {
                if (rank[rootX] > rank[rootY]) {
                    root[rootY] = rootX;
                } else if (rank[rootX] < rank[rootY]) {
                    root[rootX] = rootY;
                } else {
                    //如果深度相同且根节点不同，则新的根节点的深度+1
                    root[rootY] = rootX;
                    rank[rootX] += 1;
                }
                count--;
            }
        }

        int getCount() {
            return count;
        }
    }

                                                                                                                                                                                                                                          /**
     * 长度最小的子数组：滑动窗口
     * @param s
     * @param nums
     * @return
     */
    public int minSubArrayLen01(int s, int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        int ans = Integer.MAX_VALUE;
        for (int i = 0; i < nums.length; i++) {
            int sum = 0;
            for (int j = i + 1; j < nums.length; j++) {
                sum += nums[j];
                if (sum >= s) {

                    break;
                }
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }

     * @param s

    public int minSubArrayLen02(int s, int[] nums) {

            return 0;

        }

     * 方式一：暴力解法
     * @param str
     * @return
     */
    public static int lengthOfLongestSubstring01(String str) {
        if (str.length() == 0) {
            return 0;
        }
        int ans = 0;
        HashSet<Character> set = new HashSet<>();
        for (int i = 0; i < str.length(); i++) {
            set.add(str.charAt(i));
            for (int j = i + 1; j < str.length(); j++) {
                if (set.contains(str.charAt(j))) {
                    ans = Math.max(ans, j - i);
                    break;
                } else {
                    set.add(str.charAt(j));
                }
            }
            set.clear();
        }

    }

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 方式二：滑动窗口
     * @param str
     * @return
     */
    public static int lengthOfLongestSubstring02(String str) {
        if (str.length() == 0) {
            return 0;
        }
        int ans = 0;
        HashSet<Character> set = new HashSet<>();
        //右指针
        int end = 0;
        for (int i = 0; i < str.length(); i++) {
            if (i != 0) {
                set.remove(str.charAt(i-1));
            }
            //移动
            while (end < str.length() && !set.contains(str.charAt(end))) {
                set.add(str.charAt(end));
                end++;
            }
            //得到结果
            ans = Math.max(ans, end - i);
        }
        return ans;
    }

                                                                                                                                                                                                                                          /**
     * 方式一：滑动窗口
     * @param A
     * @param K
     * @return
     */
    public static int longestOnes(int[] A, int K) {
        int ans = 0;
        int left = 0, right = 0;
        int zeroNum = 0;
        //右指针
        while (right < A.length) {
            //条件
            if (A[right] == 0) {
                zeroNum++;
            }
            //左指针
            while (zeroNum > K) {
                if (A[left] == 0) {
                    zeroNum--;
                }
                left++;
            }
            ans = Math.max(ans, right - left + 1);
            right++;
        }
        return ans;
    }

                                                                                                                                                                                                                                      用于变更字符串的最大预算是 maxCost。在转化字符串时，总开销应当小于等于该预算，这也意味着字符串的转化可能是不完全的。

                                                                                                                                                                                                                                      如果你可以将 s 的子字符串转化为它在 t 中对应的子字符串，则返回可以转化的最大长度。

                                                                                                                                                                                                                                      如果 s 中没有子字符串可以转化成 t 中对应的子字符串，则返回 0。

                                                                                                                                                                                                                                      示例 2：

                                                                                                                                                                                                                                      输入：s = "abcd", t = "cdef", maxCost = 3
输出：1
解释：s 中的任一字符要想变成 t 中对应的字符，其开销都是 2。因此，最大长度为 1。

                                                                                                                                                                                                                                      输入：s = "abcd", t = "acde", maxCost = 0

解释：a -> a, cost = 0，字符串未发生变化，所以最大长度为 1

                                                                                                                                                                                                                                      方式一：滑动窗口

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 方式一：滑动窗口
     * @param s
     * @param t
     * @param maxCost
     * @return
     */
    public static int equalSubstring(String s, String t, int maxCost) {
        //抽象出问题：转换为一个数组
        int n = s.length();
        int[] diff = new int[n];
        for (int i = 0; i < n; i++) {
            diff[i] = Math.abs(s.charAt(i) - t.charAt(i));
        }
        int ans = 0;
        int start = 0, end = 0;
        int count = 0;
        while (end < n) {
            count += diff[end];
            //左指针
            while (count > maxCost) {
                count -= diff[start];
                start++;
            }
            ans = Math.max(ans, end - start + 1);
            end++;
        }
        return ans;
    }

    public static boolean isPalindromic(String str) {
        for (int i = 0; i < str.length() / 2; i++) {
            if (str.charAt(i) != str.charAt(str.length() - 1 - i)) {
                return false;
            }
        }
        return true;
    }

    /**
     * 最长回文子串：暴力解法
     * @param str
     * @return
     */
    public static String longestPalindrome(String str) {
        int max = 0;
        String ans = "";
        for (int i = 0; i < str.length(); i++) {
            for (int j = i + 1; j < str.length(); j++) {
                String s = str.substring(i, j);
                if (isPalindromic(s) && s.length() > max) {
                    max = Math.max(max, s.length());
                    ans = s;
                }
            }
        }
        return ans;
    }

                                                                                                                                                                                                                                      代码示例：

                                                                                                                                                                                                                                          /**
     * 有效的括号：栈
     * @param s
     * @return
     */
    public static boolean isValid(String s) {
        if (s.length() % 2 != 0) {
            return false;
        }
        Map<Character, Character> map = new HashMap<Character, Character>() {
            {
                put(')', '(');
                put(']', '[');
                put('}', '{');
            }
        };
        Deque<Character> stack = new LinkedList<>();
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (!stack.isEmpty()) {
                if (stack.peek() != map.get(ch)) {
                    return false;
                }
                stack.pop();
            } else {
                stack.push(ch);
            }
        }
        return stack.isEmpty();
    }
